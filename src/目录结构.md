##目录结构

```
├── index.html
├── main.js
├── assets     # 静态资源(图片,字体)
│   └── img    # 图片
│   └── font   # 字体
├── api        # 全部请求API方法
│   └── ...    
├── utils      # 全局公共方法
│   └── ...    
├── styles     # 全局公共样式
│   └── ...    
├── routes     # vue路由配置
│   └── ...    
├── components
│   ├── Home   # 文件夹下包含Home组件相关的子组件
│   ├── shared # 文件夹下包含全部公共组件
│   └── ...
└── vuex
    ├── index.js          # 我们组装模块并导出 store 的地方
    ├── actions.js        # 根级别的 action
    ├── mutations.js      # 根级别的 mutation
    └── modules           # 各个页面store模块
        ├── cart.js       # cart模块
        └── products.js   # products模块
```
