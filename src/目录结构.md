##目录结构

```
├── index.ejs  # html页面容器
├── index.js   # 项目入口文件
├── initVue.js # 初始化Vue实例
├── App.vue    # vue根组件
├── assets     # 静态资源(图片,字体)
│   └── img    # 图片文件夹
│   └── font   # 字体文件夹
├── api        # 全部请求API方法
│   └── ...    
├── utils      # 全局公共方法
│   └── ...    
├── styles     # 全局公共样式
│   └── ...    
├── routes     # vue路由配置
│   └── ...    
├── views      # 各页面顶层组件
│   └── Home   # Home顶层组件
├── components
│   ├── Home   # 文件夹下包含Home组件相关的子组件
│   ├── shared # 文件夹下包含全部公共组件
│   └── ...
└── vuex
    ├── index.js          # 我们组装模块并导出 store 的地方
    ├── root_store.js     # 根级别store(state,actions,mutations,getters)
    ├── mutation_types.js # 全部mutation的type 常量
    └── modules           # 各个模块store(state,actions,mutations,getters)
        ├── cart.js       # cart模块
        └── products.js   # products模块
```
